\begin{definition}[Probabilistic Model]
    We define a \emph{probabilistic model} \M as a pair $\left(\set{S}, \set{P}\right)$, where \set{S} is a sample
    space, and \set{P} is a set of probability measures on \set{S}.
\end{definition}

\begin{definition}[Divergence Measure]
    Let $\pairsX$ be a set of pairs of discrete random
    variables in a probabilistic model \model.
    We define the \emph{divergence measure of $P \in \set{P}$ for \set{X}} as follows:
    \begin{equation}
        \DxP = \max_{Q \in \set{P}} \Ex{Q}{\sum_i \Dcond},
        \label{eq:div_measure}
    \end{equation}
    where $\Ex{Q}{\,\cdot\,}$ denotes expectation relative to $Q$ and each $\Dcond$ is the Kullback-Leibler divergence
    between $P(\rv{X}|\rval{E})$ and $Q(\rv{X}|\rval{E})$:
    \[
        \Dcond = \sum_{\rval{x}} Q(\xIe) \ln \frac{Q(\xIe)}{P(\xIe)}\cdot
    \]
    We also define \emph{the central distribution for \set{X}} to be a distribution $C_\set{X} \in \set{P}$ which
    minimizes the divergence measure for \set{X}:
    \begin{equation}
        C_\set{X} = \arg\pmin \DxP\cdot
        \label{eq:center}
    \end{equation}
\end{definition}

\begin{proposition}
    Using the linearity of expectation, we have
    \begin{equation*}
        \DxP = \max_{Q \in \set{P}} \sumOverixe Q(\xe) \ln \frac{Q(\xIe)}{P(\xIe)}\cdot
        \label{eq:div_measure_full}
    \end{equation*}
\end{proposition}

\begin{definition}[Parametric Model]
    Let $\param$ be a random vector.
    A parametric model is a probabilistic model \pmodel, where for every $P \in \set{P}$ there is a prior probability
    distribution $p$, such that for every $\rval[]{s} \in \set{s}$ we have
    \begin{equation}
        \paramDist{\rval[]{s}},
        \label{eq:parametric}
    \end{equation}
    while $\pi$ is a known probability distribution.
\end{definition}

\begin{lemma}
    \label{lm:DxP_uppper_bound}
    Let \pmodel be a parametric model, and $\pairsX$ be a set of discrete random variable pairs in \Mt.
    For every $P \in \set{P}$ we have
    \begin{align*}
        \DxP &\leq \tmax \DExparam\\  \label{eq:DxP_upper_bound}
        &= \tmax \Dparam\cdot
    \end{align*}
\end{lemma}

\begin{proof}
    Let $R$ be an arbitrary distribution in \Mt, since the Kullback-Leibler divergence is positive,
    its expected value is also positive, and for every parameter value $\paramValue$ we have
    \[
        \Dparam[R] \geq 0\ \cdot
    \]
    Using this, for every distribution $P$ and every parameter \vval{\theta}, we obtain
    \newcommand{\RP}{\sumOverixe \pi(\xeIt) \ln \frac{R(\xIe)}{P(\xIe)}}
    \[
        \RP \leq \Dparam\cdot
    \]
    Since this inequality holds for any $\paramValue$, the maximum of the right-hand side
    is also greater than or equal to the maximum of the left-hand side w.r.t. $\paramValue$:
    \begin{equation}
        \tmax \RP \leq \tmax \DExparam\cdot
        \label{eq:ineq-theorem1}
    \end{equation}

    \Mt is a parametric model and $\paramDist[R]{\xe}$, where $r(\paramValue)$ is
    a probability distribution.
    Clearly, for every $\paramValue$ we have
    \[
        \RP \leq \tmax \RP\cdot
    \]
    Since $r$ is a distribution and is non-negative, we can multiply this inequality
    by $r(\paramValue)$ and integrate.
    Notice that the right hand side is not a function of $\paramValue$ and it will remain unchanged
    because $\int \dt[r] = 1$:
    \[
        \Ex{R}{\sum_i \Dcond[R]} \leq \tmax \RP\cdot
    \]
    Now from Inequality~\ref{eq:ineq-theorem1}, for \emph{any} distribution $P$ and $R$ in \Mt, it follows that
    \[
        \Ex{R}{\sum_i \Dcond[R]} \leq \tmax \DExparam,
    \]
    which proves the theorem.
\end{proof}

\begin{theorem}
    Let \pmodel be a parametric model and $\set{X}$ be a set of pairs of discrete random variables in \Mt.
    If for every value of $\paramValue$ there is a distribution $R \in \set{P}$ such that
    $R(\rval[]{s}) = \pi(\rval[]{s} \mid \paramValue)$ for every $\rval[]{s} \in \set{S}$,
    then for any $P \in \set{P}$ we have:
    \begin{equation}
        \DxP = \tmax \DExparam\cdot
        \label{eq:div_parametric}
    \end{equation}
    \label{th:div_for_parametric}
\end{theorem}

\begin{proof}
    For an arbitrary distribution $P \in \set{P}$, we let
    \[
        \paramValue^* = \arg\tmax \DExparam\cdot
    \]
    In \Mt there should be a distribution $R \in \set{P}$ that equals $\pi(\cdot \mid \paramValue^*)$, and we can write
    \[
        \Ex{R}{\sum_i \Dcond[R]} = \DExparam[\paramValue^*]\cdot
    \]
    On the other hand, we see
    \[
        \max_{Q \in \set{P}} \Ex{Q}{\sum_i \Dcond} \geq \Ex{R}{\sum_i \Dcond[R]}\cdot
    \]
    Thus,
    \[
        \DxP \geq \tmax \DExparam\cdot
    \]
    This inequality and Lemma \ref{lm:DxP_uppper_bound} prove the theorem.
\end{proof}

\begin{theorem}[Minimax theorem~\cite{Du1995}]
    \label{th:minimax}
    Let $\set{X} \subset \mathbb {R}^{n}$ and $\set{Y} \subset \mathbb {R}^{m}$ be compact convex sets.
    If $f:\set{X}\times \set{Y}\rightarrow \mathbb {R}$ is a continuous function that is concave-convex, i.e.

    \begin{itemize}
        \item $f(\cdot ,y):\set{X}\to \mathbb {R}$ is concave for fixed $y$, and
        \item $f(x,\cdot ):\set{Y}\to \mathbb {R}$ is convex for fixed $x$.
    \end{itemize}
    Then we have that
    \begin{equation}
        \max_{x\in \set{X}}\min_{y\in \set{Y}}f(x, y) = \min_{y\in \set{Y}}\max_{x\in \set{X}}f(x,y)\cdot
        \label{eq:minimax}
    \end{equation}
\end{theorem}

\begin{theorem}
    \label{th:central_by_max}
    Consider a generic parametric model \pmodel,
    and let $\pairsX$ be a set of pairs of discrete random variables in \Mt.
    Assume that for every $Q_1, Q_2 \in \set{P}$ and any real number $0 \leq \alpha \leq 1$, there is
    a distribution $Q^* \in \set{P}$, such that for every $i, \xe$ we have
    \begin{equation}
        Q^*(\xIe) = \alpha Q_1(\xIe) + (1 - \alpha) Q_2(\xIe)\cdot
        \label{eq:convex_dist_set}
    \end{equation}
    In this model, the central distribution for $\set{X}$ can be obtained by:
    \begin{align}
        \Cx &= \arg\pmax \int \DExparam \dt\\
        &= \arg\pmax \Dminimax,
        \label{eq:Cx_by_max}
    \end{align}
    where $P(\xet) = \pi(\xeIt) p(\paramValue)$.
\end{theorem}

\begin{proof}
    \newcommandx{\f}[2][1 = r, 2 = P_{\rv{X} \mid \rval{e}}]{ f(#1, #2)}
    We prove this theorem using the Minimax theorem.
    For that, we define a functional as follows:
    \begin{equation*}
        \f = \int \DExparam \dt[r],
    \end{equation*}
    where $r(\paramValue)$ is an arbitrary probability distribution, and $P \in \set{P}$.
    Since $r(\paramValue)$ can be any distribution, maximizing with respect to $r$ is like maximizing
    with respect to $\paramValue$, and we have
    \[
        \Cx = \arg\pmin \max_{r} \f\cdot
    \]

    The domain of $r$ is the set of all probability distributions, which is a compact convex set.
    Moreover, from Equation~\ref{eq:convex_dist_set}, it follows that the set of probability distributions for
    $\rv{X}|\rv{E}$ in \Mt is also a compact convex set.
    Thus, the domain of $\f$ is a compact convex set.

    For a fixed $P_{\rv{X} \mid \rval{e}}$, $\f[\cdot]$ is a linear function, which is both concave and convex.
    On the other hand, if we let $\paramDist[r]{\xe}$ for $\f[r][\cdot]$ we have
    \begin{equation}
        \f[r][\cdot] = c - \sumOverixe R(\xe) \ln P(\xIe)\cdot
        \label{eq:max_for_P}
    \end{equation}
    If we calculate the Hessian matrix of $\f[r][\cdot]$, we observe that the matrix is diagonal, and each element
    of the main diagonal is $\frac{R(\xIe)}{P(\xIe)^2}$.
    Thus, the hessian matrix of $\f[\cdot]$ is positive semi-definite and $\f[r][\cdot]$ is a convex function.
    Now, Theorem~\ref{th:minimax} implies
    \begin{equation}
        \pmin \max_{r} \f = \max_{r} \pmin \f\cdot
        \label{eq:divergence_minmax}
    \end{equation}
    From Equation~\ref{eq:max_for_P} we can see that for an arbitrary fixed $r$ we have
    \[
        \pmin \f = \pmax \sumOverixe R(\xe) \ln P(\xIe)\cdot
    \]

    We know that the Kullback–Leibler divergence, and therefore its expected value, is non-negative:
    \[
        \Ex{R}{\sum_i \Dcond[R]} \geq 0\cdot
    \]
    By simple algebra we get
    \[
        \sumOverixe R(\xe) \ln P(\xIe) \leq \sumOverixe R(\xe) \ln R(\xIe)\cdot
    \]
    Recall that \Mt is a generic parametric model, so $R \in \set{P}$, and we can reach this
    upper bound by letting $P = R$:
    \[
        \pmin \f = \int \Dminimax[R] d\paramValue\cdot
    \]
    By Equation~\ref{eq:divergence_minmax} we conclude
    \[
        \Cx = \arg\max_{R \in \set{R}} \int \DExparam[][R] \dt[r]\cdot
    \]
\end{proof}

\begin{theorem}
    \label{th:lambda}
    Under the assumptions of Theorem~\ref{th:central_by_max}, the central distribution for \set{X} satisfies:
    \[
        \Dparam[\Cx] = \lambda,
    \]
    where $\lambda$ is a constant.
\end{theorem}
\begin{proof}
    \renewcommand{\c}{c(\paramValue)}

    Since \Mt is a parametric model, the central distribution for \set{X} maximizes the right-hand side of
    Equation~\ref{eq:Cx_by_max} with respect to different choices of the prior distribution $p(\paramValue)$,
    subject to the constraint $\int \dt[p] = 1$.
    Applying the Lagrange multiplier technique, the local maximum of the right-hand side of
    Equation~\ref{eq:Cx_by_max} will be a stationary point of the Lagrangian function:
    \[
        \Lagr = \Dminimax - \lambda \left( \int \dt - 1 \right)\cdot
    \]
    Let us denote the prior distribution of \Cx by $\c$.
    Let $p(\paramValue) = \c + \epsilon \eta(\paramValue)$ be a variation of $\c$, where $\eta(\paramValue)$ is
    a continuously differentiable function with $\eta(0) = \eta(1) = 0$, and $\epsilon$ is a small constant.

    Since $\c$ is an extremizing function for the right-hand side of Equation~\ref{eq:Cx_by_max},
    $\Lagr[\epsilon]$ has a stationary point where $\epsilon = 0$, and we have
    \[
        \parDerivAt = 0\cdot
    \]
    Note that
    \begin{align*}
        \de{p(\paramValue)} &= \eta(\paramValue),\\
        \de{P(\xe)} &= \int \pi(\xeIt) \dt[\eta],\\
        \de{P(\rval{e})} &= \int \pi(\eIt) \dt[\eta]\ \cdot
    \end{align*}
    Using the chain rule, we calculate the partial derivative of $\Lagr[\epsilon]$ relative to $\epsilon$:
    \begin{align*}
        \parDeriv\ =\ &\int \left( \Dparam  \right) \dt[\eta] \\
        &- \int \left( \sumOverixe \pi(\xeIt) \frac{\int \pi(\xeIt) \dt[\eta]}{P(\xe)} \right) \dt \\
        &+ \int \left( \sum_i \sum_{\rval{e}} \pi(\eIt) \frac{\int \pi(\eIt) \dt[\eta]}{P(\rval{e})} \right) \dt \\
        &- \int \lambda \dt[\eta]
    \end{align*}
    Since $\paramDist{\xe}$ and $\paramDist{\rval{e}}$ we get
    \begin{equation*}
        \parDeriv = \int \left(-\lambda + \Dparam \right) \dt[\eta]\cdot
    \end{equation*}
    We let $ \parDerivAt = 0$.
    Since $\eta(\paramValue)$ is an arbitrary function, by applying the fundamental lemma of the calculus of
    variations and letting $\epsilon = 0$, we have
    \[
        \Dparam[\Cx] = \lambda\cdot \qedhere
    \]
\end{proof}
