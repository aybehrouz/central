\begin{definition}[Probabilistic Model]
    We define a \emph{probabilistic model} \M as a pair $\left(\set{S}, \set{P}\right)$, where \set{S} is a sample
    space, and \set{P} is a set of probability measures on \set{S}.
\end{definition}
\begin{definition}[$\epsilon$-convergence]
    Let \rvec{x} and \rvec{e} be two random vectors in a probabilistic model \model.
    We say \xIe (\rvec{x} given \rvec{e}) \emph{$\epsilon$-converges} with a distribution $P$ in \M, if for every
    distribution $Q \in \set{p}$ we have:
    \begin{equation}
        \KLcond < \epsilon \ \cdot
        \label{eq:e-convergence}
    \end{equation}
    We call $P$ an \emph{$\epsilon$-convergent distribution} for \xIe in \M.
\end{definition}

\begin{definition}[Convergence Measure]
    Let \rvec{x} and \rvec{e} be random vectors in a probabilistic model \model.
    We define the \emph{convergence measure} of \xIe with $P \in \set{P}$ as follows:
    \begin{equation}
        \measure{C}_P \left[\xIe \right] = \max_{Q \in \set{P}} \KLcond\ \cdot
        \label{eq:Cp}
    \end{equation}
    And, we define the convergence measure of \xIe in \M as follows:
    \begin{equation}
        \measure{C}_\M \left[\xIe \right] = \pmin \measure{C}_P \left[\xIe \right]\ \cdot
        \label{eq:Cm}
    \end{equation}
\end{definition}

\begin{proposition}
    If $\measure{C}_\M \left[\xIe \right] \geq \epsilon$, then there is no $\epsilon$-convergent distribution
    for \xIe in \M.
\end{proposition}

\begin{definition}[Parametric Model]
    Let \rvec{\theta} be a random vector.
    A parametric model is a probabilistic model \pmodel, where for
    every $P \in \set{P}$ there is a prior probability $p$, such that for every $\rval{s} \in \set{s}$ we have:
    \begin{equation}
        P(\rval{s}) = \int \pi(\rval{s} | \vval{\theta}) p(\vval{\theta}) d\theta \ ,
        \label{eq:parametric}
    \end{equation}
    while $\pi$ is some known probability distribution.
\end{definition}

\begin{theorem}
    Let \pmodel be a parametric model, and \rvec{x}, \rvec{e} be two random vectors in \Mt.
    For every $P \in \set{P}$ we have:
    \begin{equation}
        \measure{C}_P \left[\xIe \right] < \tmax \KLparam\ \cdot
        \label{eq:parametric_Cp}
    \end{equation}
    Moreover, by bayes' theorem we have:
    \begin{equation}
        \frac{\pi(\vval{x \mid e, \theta})}{P(\vval{x \mid e})} =
        \frac{P(\vval{\theta \mid x, e})}{P(\vval{\theta \mid e})} \ \cdot
        \label{eq:bayes}
    \end{equation}
\end{theorem}

\begin{proof}
    Let $R$ be an arbitrary distribution in \Mt, since the Kullback-Leibler divergence is always positive,
    its expected value is also positive and for every parameter value \rvec{\theta} we have:
    \[
        \KLparam[R] \geq 0 \ \cdot
    \]
    Using this, we can derive the following inequality for every distribution $P$ and
    every parameter \vval{\theta}:
    \[
        \KLparam[P][R][x][e][e] \leq \KLparam \ \cdot
    \]
    Since this inequality holds for any value of \vval{\theta}, the maximum of the right-hand side
    is also greater than or equal to the maximum of the left-hand side w.r.t \vval{\theta}:
    \begin{equation}
        \tmax \KLparam[P][R][x][e][e] \leq \tmax \KLparam \ \cdot
        \label{eq:ineq-theorem1}
    \end{equation}

    \Mt is a parametric model and
    $
    \prob [R]{x,e} = \int \pcond [\pi]{x,e}{\theta} \prob [r]{\theta} d\vval{\theta}
    $, while $\prob [r]{\theta}$ is a probability distribution.
    Obviously for every \vval{\theta} we have:
    \[
        \KLparam[P][R][x][e][e] \leq \tmax \KLparam[P][R][x][e][e]\ \cdot
    \]
    Since $\prob [r]{\theta}$ is a distribution and is non-negative, we can multiply both sides of this inequality
    by $\prob [r]{\theta}$ and integrate.
    Notice that the right hand side is not a function of \vval{\theta} and it remains unchanged
    because $\int \prob [r]{\theta} d\theta = 1$:
    \[
        \KLcond[P][R] \leq \tmax \KLparam[P][R][x][e][e]\ \cdot
    \]
    Now by Inequality~\ref{eq:ineq-theorem1}, for every distribution $P$ and $R$ we conlcude:
    \[
        \KLcond[P][R] \leq \tmax \KLparam\ ,
    \]
    which proves the theorem.
\end{proof}

\begin{corollary}

    \end{equation}

\end{corollary}

\begin{theorem}[Minimax~\cite{Du1995}]
    Let $X \subset \mathbb {R}^{n}$ and $Y \subset \mathbb {R} ^{m}$ be compact convex sets.
    If $f:X\times Y\rightarrow \mathbb {R}$ is a continuous function that is concave-convex, i.e.

    \begin{itemize}
        \item $f(\cdot ,y):X\to \mathbb {R}$ is concave for fixed $y$, and
        \item $f(x,\cdot ):Y\to \mathbb {R}$ is convex for fixed $x$.
    \end{itemize}
    Then we have that:
    \begin{equation}
        \max_{x\in X}\min_{y\in Y}f(x,y) = \min_{y\in Y}\max_{x\in X}f(x,y)\ \cdot
        \label{eq:minimax}
    \end{equation}
\end{theorem}

\begin{definition}[$\epsilon$-convergent distribution]
    Let $S$ be a random variable, and \set{E} be a set of random variables in a probabilistic model $\M$.
    We say $P$ is an \emph{$\epsilon$-convergent distribution} for $\given{S}{\set{E}}$ in $\M$, if for all
    $E \in \set{E}$, $\given{S}{E}$ $\epsilon$-converges by $P$ in $\M$.
\end{definition}


\begin{proposition}
    Let $S$ be a random variable, and $\set{E}$ be a set of random variables in a probabilistic model $\M$.
    If there exists $E \in \set{E}$ such that $\measure{C}_\M[\given{S}{E}] \geq \epsilon$, then there is no
    $\epsilon$-convergent distribution for $\given{S}{\set{E}}$ in $\M$,
\end{proposition}

\begin{theorem}
    Consider two probabilistic models $\M$ and $\aug{\M}$.
    Let $\set{E}$ be a countable set of random variables in $\M$, and $\aug{E}$ be a single random variable
    in $\M^*$.
    Suppose that there exists a random variable $I$ in $\aug{\M}$, such that for \textbf{every} probability measure $P$
    in $\M$, there exists a probability measure $\aug{P}$ in $\aug{\M}$,
    such that for every $E_i \in \set{E}$, every $s \in \range{s}$ and every $e_i \in \range{E}_i$, there exist
    some values $\aug{s}, \aug{e}$ such that we have:
    \begin{equation}
        P(S=s,E_i=e_i) = \aug{P}(\aug{S}=\aug{s},\aug{E}=\aug{e} \mid I = i)\ \cdot
        \label{eq:th1-1}
    \end{equation}
    In addition, $I$ and $\aug{E}$ are conditionally independent:
    \begin{equation}
        \aug{P}(\aug{S} \mid \aug{E}) = \aug{P}(\aug{S} \mid \aug{E}, I)\ \cdot
        \label{eq:th1-2}
    \end{equation}
    Then $\given{S}{\set{E}}$ $\epsilon$-converges by $P$ in $\M$,
    if $\given{\aug{S}}{\aug{E}}$ $\epsilon$-converges by $\aug{P}$ in $\aug{\M}$.
\end{theorem}

\begin{proof}
    Assume, by contradiction, that $\given{S}{\set{E}}$ does not $\epsilon$-converge by $P$ in $\M$.
    Therefore, there must exist $Q$ and $E_i \in \set{E}$ in $\M$ such that:
    \[
        \ED{P_{\given{S}{E_i}}}{Q_{\given{S}{E_i}}}{Q} \geq \epsilon \ \cdot
    \]
    By using equations~\ref{eq:th1-1} and~\ref{eq:th1-2} we can rewrite the left hand side:
    \begin{align*}
        \ED{P_{\given{S}{E_i}}}{Q_{\given{S}{E_i}}}{Q} &= \sumover{s} \sum_{e_i \in \mathcal{E}_i} \PQdive{s}{e_i} \\
        &= \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e}, I = i)}{\aug{P}(\aug{s} \mid \aug{e}, I = i)}\aug{Q}(\aug{s},\aug{e} \mid I = i)\\
        &= \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})}\aug{Q}(\aug{s},\aug{e} \mid I = i) \ \cdot
    \end{align*}
    Hence, we have:
    \[
        \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})}\aug{Q}(\aug{s},\aug{e} \mid I = i) \geq \epsilon \ \cdot
    \]
    Since $\aug{Q}(i)$ is a distribution and is non-negative for every $i$, we can multiply both sides of this
    inequality to $\aug{Q}(i)$ and sum over all values of $i$:
    \begin{align*}
        \sum_{i} \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})}\aug{Q}(\aug{s},\aug{e}, I = i)
        &\geq \sum_{i} \epsilon \aug{Q}(i) \\
        \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})} \sum_{i}  \aug{Q}(\aug{s},\aug{e}, I = i)
        &\geq \epsilon \sum_{i}  \aug{Q}(i) \\
        \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})}\aug{Q}(\aug{s},\aug{e})
        &\geq \epsilon \ ,
    \end{align*}
    which means $\given{\aug{S}}{\aug{E}}$ does not $\epsilon$-converge by $\aug{P}$ in $\aug{\M}$.
\end{proof}

\begin{definition}[conditional model]
    Let $\M \equiv \left(\set{S}, \set{P}\right)$ be a probabilistic model. $\M$ conditioned on an event
    $\set{A} \subset \set{S}$ is a probabilistic model $\M_{|\set{A}} \equiv (\set{A}, \set{P}_{|\set{A}})$,
    where $\set{P}_{|\set{A}}$ is the set of probability measures obtained by conditioning every probability
    measure in $\set{P}$ on $\set{A}$.
\end{definition}

\begin{definition}[model radius]
    Let $\M \equiv \left(\set{S}, \set{P}\right)$ be a probabilistic model.
    We define the \emph{radius} of $\M$ as follows:
    \begin{equation}
        \measure{R}_\M = \minmax \left\{ \KL{Q}{P} \right\} = \minmax \sumover{s} Q(s)\ln \frac{Q(s)}{P(s)}\ \cdot
        \label{eq:radius}
    \end{equation}
\end{definition}
