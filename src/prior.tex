\begin{definition}[probabilistic model]
    We define a \emph{probabilistic model} as a pair $(\mathcal{S}, \mathcal{P})$, where $\mathcal{S}$ is a sample
    space, and $\mathcal{P}$ is a set of \emph{acceptable} probability measures on $\mathcal{S}$.
\end{definition}

\begin{definition}[$\epsilon$-convergence]
    Let $S$ and $E$ be random variables in a probabilistic model $\M$.
    We say $\given{S}{E}$ ($S$ given $E$) \emph{$\epsilon$-converges} by a distribution $P$ in $\M$,
    if for every distribution $Q$ in $\M$ we have:
    \begin{equation}
        \ED{P_{\given{S}{E}}}{Q_{\given{S}{E}}}{Q} =
        \sumover{s}\sumover{e} \Dexpand{s}{e} < \epsilon \ \cdot
        \label{eq:e-convergence}
    \end{equation}
\end{definition}

\begin{definition}[$\epsilon$-convergent distribution]
    Let $S$ be a random variable, and $\mathbf{E}$ be a set of random variables in a probabilistic model $\M$.
    We say $P$ is an \emph{$\epsilon$-convergent distribution} for $\given{S}{\mathbf{E}}$ in $\M$, if for all
    $E \in \mathbf{E}$, $\given{S}{E}$ $\epsilon$-converges by $P$ in $\M$.
\end{definition}

\begin{theorem}
    Consider two probabilistic models $\M$ and $\aug{\M}$.
    Let $\mathbf{E}$ be a countable set of random variables in $\M$, and $\aug{E}$ be a single random
    variable in $\M^*$.
    Assume there exist a random variable $I$ in $\aug{\M}$, such that for every probability measure $Q$
    in $\M$, there exist a probability measure $\aug{Q}$ in $\aug{\M}$,
    such that for every $E_i \in \mathbf{E}$ we have:
    \begin{equation}
        Q(S,E_i) = \aug{Q}(\aug{S},\aug{E} \mid I = i)\ ,
        \label{eq:th1-1}
    \end{equation}
    while $I$ and $\aug{E}$ are conditionally independent:
    \begin{equation}
        \aug{Q}(\aug{S} \mid \aug{E}) = \aug{Q}(\aug{S} \mid \aug{E}, I)\ \cdot
        \label{eq:th1-2}
    \end{equation}
    Then $\given{S}{\mathbf{E}}$ $\epsilon$-converges by $P$ in $\M$,
    if $\given{\aug{S}}{\aug{E}}$ $\epsilon$-converges by $\aug{P}$ in $\aug{\M}$.
\end{theorem}

\begin{proof}
    Assume, by contradiction, that $\given{S}{\mathbf{E}}$ does not $\epsilon$-converge by $P$ in $\M$.
    Therefore, there must exist $Q$ and $E_i \in \mathbf{E}$ in $\M$ such that:
    \[
        \ED{P_{\given{S}{E_i}}}{Q_{\given{S}{E_i}}}{Q} \geq \epsilon \ \cdot
    \]
    Equations~\ref{eq:th1-1} and~\ref{eq:th1-2} hold for every $Q$ and $E_i$, and we can rewrite the left hand
    side of this inequality using them:
    \begin{align*}
        \ED{P_{\given{S}{E_i}}}{Q_{\given{S}{E_i}}}{Q} &= \sumover{s} \sum_{e_i \in \mathcal{E}_i} \Dexpand{s}{e_i} \\
        &= \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e}, I = i)}{\aug{P}(\aug{s} \mid \aug{e}, I = i)}\aug{Q}(\aug{s},\aug{e} \mid I = i)\\
        &= \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})}\aug{Q}(\aug{s},\aug{e} \mid I = i) \ ,
    \end{align*}
    so we must have:
    \[
        \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})}\aug{Q}(\aug{s},\aug{e} \mid I = i) \geq \epsilon \ \cdot
    \]
    Since $\aug{Q}(i)$ is a distribution and is non-negative for every $i$, we can multiply both sides of this
    inequality to $\aug{Q}(i)$ and sum over all values of $i$.
    \begin{align*}
        \sum_{i} \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})}\aug{Q}(\aug{s},\aug{e}, I = i)
        &\geq \sum_{i} \epsilon \aug{Q}(i) \\
        \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})} \sum_{i}  \aug{Q}(\aug{s},\aug{e}, I = i)
        &\geq \epsilon \sum_{i}  \aug{Q}(i) \\
        \sumover{\aug{s}}\sumover{\aug{e}}\ln \frac{\aug{Q}(\aug{s} \mid \aug{e})}{\aug{P}(\aug{s} \mid \aug{e})}\aug{Q}(\aug{s},\aug{e})
        &\geq \epsilon \ ,
    \end{align*}
    which means $\given{\aug{S}}{\aug{E}}$ does not $\epsilon$-converge by $\aug{P}$ in $\aug{\M}$.
\end{proof}
