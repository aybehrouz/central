With the concept of hierarchical central distributions in place, we now develop the theoretical machinery needed
to characterize and compute them.
Throughout this subsection, $X$ denotes a discrete random variable with finite support and \setQ denotes a finite
set of probability distributions over $X$.

As we shall see, central distributions arise naturally as solutions to a convex--concave optimization problem.
By formulating the appropriate objective function, we obtain a saddle-point characterization that enables
the use of standard techniques from convex optimization.

\begin{definition} [Saddle Point]
    Consider a function $f:\set{w} \times \set{z} \to \field{R}$.
    We refer to a pair $(w^*, z^*)$ as a \emph{saddle point} for $f$ if we have
    \[
        f(w^*, z) \leq f(w^*, z^*) \leq f(w, z^*),
    \]
    for all \inSet{w} and \inSet{z}.
\end{definition}

\begin{lemma} \label{lm:saddle-exists}
    Define:
    \begin{equation*}
        \objf = \sum_i \klx[Q_i] R(i),
    \end{equation*}
    where $P$ and $R$ are probability distributions, $Q_i \in \setQ$ and the index $i$ ranges over all elements of~\setQ.
    Then there exists a saddle point for~$\objf[]$.
\end{lemma}

\begin{proof}
    The Kullback--Leibler divergence is convex and lower-semicontinuous in its second argument.
    Therefore, for any fixed probability distribution $R$,
    the function $\objf[\none][R]$ is a finite convex combination of
    lower-semicontinuous convex functions and is therefore itself convex and lower-semicontinuous in $P$.

    For fixed $P$, the function $\objf[P][\none]$ is affine in $R$, and hence concave and continuous.

    The set of probability distributions over a discrete random variable with finite support is a compact convex
    simplex.
    Therefore, the assumptions of Sion's minimax theorem are satisfied~\cite{10.2996/kmj/1138038812}, and
    \[
        \min_P \sup_R \objf = \sup_R \min_P \objf.
    \]
    Since $\objf[P][\none]$ is continuous and the feasible set for $R$ is compact,
    the supremum is attained for every $P$.
    Hence the minimax solution exists and $\objf[]$ admits a saddle point.
\end{proof}

\begin{lemma} \label{lm:saddle-is-central}
    % I've been unable to prove that the central distribution is a saddle point for f, or characterize conditions
    % that imply that.
    Let $f$ be the function defined in Lemma~\ref{lm:saddle-exists}.
    If \saddle is a saddle point for $f$, then \saddP is a \cd.
\end{lemma}

\begin{proof}
    \newcommand{\fsaddle}{\objf[\saddP][\saddR]}
    Let $P$ be an arbitrary probability distribution over $X$, and \saddle be a saddle point for $\objf[]$.
    As $P$ is a probability distribution, $\objf[P][\saddR]$ is defined, and by the definition of a saddle point
    \begin{equation}
        \label{eq:th_proof_saddle_lq_dx}
        \fsaddle \leq \objf[P][\saddR].
    \end{equation}
    Since \saddR is a probability distribution, the definition of $f$ implies:
    \[
        \objf[P][\saddR] \leq \sFunc[P].
    \]

    On the other hand, by the definition of a saddle point:
    \[
        \fsaddle = \max_{R} \objf[\saddP][R] = \sFunc[\saddP].
    \]
    Hence, Equation~\ref{eq:th_proof_saddle_lq_dx} implies
    \[
        \sFunc[\saddP] \leq \sFunc[P].
    \]
    This holds for any arbitrary $P$.
    Therefore, by Definition~\ref{def:center-of-set}, $\saddP$ is a \cd.
\end{proof}

\begin{lemma}
    \label{lm:saddle_conditions}
    Let $f$ be the function defined in Lemma~\ref{lm:saddle-exists}.
    The point \saddle is a saddle point for $f$, if and only if there is a constant $\lambda$, and a
    function $\mu$, such that for every $x$ and $i$ the following conditions hold:
    \begin{gather}
        \label{eq:kkt_lambda}
        \klExp[Q_i][\saddP][x] + \mu(i) = \lambda, \\
        \label{eq:kkt_w_avg}
        \saddP(x) = \sum_i \saddR(i) Q_i(x) \\
        \label{eq:kkt_r_sums_one}
        \sum_i \saddR(i) = 1, \\
        \label{eq:kkt_r_geq_0}
        \saddR(i) \geq 0, \\
        \label{eq:kkt_mu_geq_zero}
        \mu(i) \geq 0, \\
        \label{eq:kkt_rmu_is_zero}
        \mu(i) \saddR(i) = 0.
    \end{gather}

    Furthermore, for a fixed probability distribution $R$, a distribution $\hat{P}$ minimizes $f(\none,R)$, if and
    only if for all $x$ it satisfies
    \begin{equation}
        \label{eq:saddle_lm_minimizer_dist}
        \hat{P}(x) = \sum_i Q_i(x)R(i).
    \end{equation}
\end{lemma}

\begin{proof}
    We extend $\objf[]$ to a function $\hat{f}$ defined as follows:
    \begin{equation*}
        \hat{f}(P, R) = \fExp,
    \end{equation*}
    where the domain is extended by relaxing the constraints that $P$ and $R$ be probability distributions.
    This enables us to formulate the saddle points of $\objf[]$ as solutions to a constrained optimization problem
    defined for~$\hat{f}$.

    Let \saddle be a simultaneous solution to the following two optimization problems:

    \begin{equation}
        \begin{aligned}
            &\text{minimize} & &\hat{f}(P,R) & &\text{w.r.t.} \ P \\
            &\text{subject to} & &\sum_{x} P(x) = 1, \\
            \\
            &\text{maximize} & &\hat{f}(P,R) & &\text{w.r.t.} \ R \\
            &\text{subject to} & &R(i) \geq 0, & &\text{for all} \ i  \\
            & & &\sum_{i} R(i) = 1 &
        \end{aligned}\label{eq:twin_optimization}
    \end{equation}
    If $\saddle$ exists, it will be a saddle point for $\objf[]$.

    Both optimization problems are convex and differentiable; see~\cite{boyd2004convex}.
    To establish this, consider that all the constraint functions are affine.
    Additionally, both $\hat{f}(P,\none)$ and $\hat{f}(\none, R)$ are differentiable on their domains.
    For any fixed $P$, $\hat{f}(P,\none)$ is an affine function and is concave.
    For a fixed solution to the second optimization problem denoted by $\hat{R}$, we can write
    \begin{equation}
        \label{eq:simplified_f}
        \hat{f}(P,\hat{R}) = c - \sum_{i,x} \hat{R}(i) Q_i(x) \ln P(x),
    \end{equation}
    where $c$ is not a function of $P$.
    Since $\hat{R}$ is a solution to the second optimization problem, it satisfies the corresponding constraints and
    is therefore a valid probability distribution.
    Hence, for all $i$, $x$ we have $\hat{R}(i)Q_i(x) \geq 0$, and $\hat{f}(\none, \hat{R})$  is
    the negative of a linear combination of concave $\ln P(x)$ functions with nonnegative coefficients.
    This function is therefore convex.

    Thus, both optimization problems are convex and differentiable, with affine equality and inequality constraints.
    In this setting, Slater's condition reduces to feasibility~\cite{boyd2004convex}.
    The feasible set for the first problem is the probability simplex, while the feasible set for the second problem is
    the strictly positive probability simplex.
    Both sets are nonempty.
    Therefore, Slater's condition holds, and the KKT conditions are necessary and sufficient for optimality.
    In other words, if there exists a point $\saddle$ that satisfies the KKT conditions for both problems, it will be a
    saddle point for $\objf[]$, and vice versa.

    The KKT conditions are as follows.
    There exist constants $\lambda$, $\eta$ and a function $\mu$, such that for every $x$ and $i$:
    \begin{gather}
        \label{eq:kkt_proof_R_is_pxIe}
        \sum_i \saddR(i) Q_i(x) = \eta \saddP(x) \\
        \label{eq:kkt_proof_p_sums_one}
        \sum_{x} \saddP(x) = 1, \\
        \nonumber
        \klExp[Q_i][\saddP][x] + \mu(i) = \lambda, \\
        \label{eq:kkt_proof_r_sums_one}
        \sum_i \saddR(i) = 1, \\
        \nonumber
        \saddR(i) \geq 0, \\
        \nonumber
        \mu(i) \geq 0, \\
        \nonumber
        \mu(i) \saddR(i) = 0.
    \end{gather}
    By summing Equation~\ref{eq:kkt_proof_R_is_pxIe} over $x$ and using Equation~\ref{eq:kkt_proof_r_sums_one},
    we obtain $\eta = 1$.
    Substituting this into Equation~\ref{eq:kkt_proof_R_is_pxIe} yields Equation~\ref{eq:kkt_w_avg},
    which in turn implies Equation~\ref{eq:kkt_proof_p_sums_one}.

    To prove the second part of the lemma, we note that Equations~\ref{eq:kkt_proof_R_is_pxIe}
    and~\ref{eq:kkt_proof_p_sums_one} are the KKT conditions for the minimization problem
    in Equation~\ref{eq:twin_optimization}.
    As shown above, the KKT conditions are necessary and sufficient for optimality,
    and when $R$ is a probability distribution, these conditions simplify to
    \[
        \hat P(x)=\sum_i Q_i(x)R(i). \qedhere
    \]
\end{proof}

\begin{lemma}
    \label{lm:saddle-is-maximizer}
    Let $f$ be the function defined in Lemma~\ref{lm:saddle-exists}.
    Define the function $g$ as:
    \[
        g(R) = \sum_i \klx[Q_i][R] R(i),
    \]
    where $Q_i \in \setQ$, and $R$ is a probability distribution such that
    \[
        R(X, I) = Q_I(X)R(I).
    \]
    If \saddle is a saddle point for $f$, then the joint distribution $Q_I(X)\saddR(I)$ is a maximizer of $g$.
\end{lemma}

\begin{proof}
    Let \saddle be a saddle point for $f$.
    By Lemma~\ref{lm:saddle_conditions}, \saddP satisfies Equation~\ref{eq:kkt_w_avg}.
    Letting $\saddR(X, I) = Q_I(X) \saddR(I)$ we have
    \[
        g(\saddR) = f\saddle.
    \]
    Now, let $R$ be an arbitrary probability distribution such that $g(R)$ is defined.
    From the second part of Lemma~\ref{lm:saddle_conditions}, it follows that $g(R)$ indeed corresponds to the
    minimum of $f(\none, R)$, which implies
    \[
        g(R) \leq f(\saddP,R).
    \]

    Since \saddle is a saddle point, we also have
    \[
        f(\saddP, R) \leq f\saddle.
    \]
    Combining the two inequalities, we get:
    \[
        g(R) \leq g(\saddR).
    \]
    This shows that $g(\saddR)$ is the maximum of $g$, as desired.
\end{proof}

The following theorem establishes the existence and uniqueness of the central distribution and provides a
characterization of its optimality in the KL-minimax setting.
This characterization motivates the term \emph{central distribution}:
like the center of a circle, the central distribution is equidistant from the boundary.

\begin{theorem} [Centrality]
    \label{th:centrality}
    Let $X$ be a discrete random variable with finite support, and let \setQ be a finite set of probability
    distributions over $X$.
    Then \setQ has a unique central distribution for $X$.

    Moreover, \cDist is the \cd\ if and only if it can be expressed as a convex combination of the elements
    of \setQ; that is,
    \[
        \cDist(x) = \sum_i Q_i(x) R(i),
    \]
    where $R$ is a probability distribution over the indices of $Q$,
    and there exists a constant $\lambda$ such that
    \begin{equation*}
        \klx[Q_i][\cDist] = \lambda
    \end{equation*}
    whenever $R(i) > 0$, and
    \[
        \klx[Q_i][\cDist] \leq \lambda
    \]
    whenever $R(i) = 0$.
\end{theorem}

\begin{proof}
    By Lemma~\ref{lm:saddle-exists} the function $\objf[]$ has a saddle point.
    Let $(\cDist, R)$ denote this saddle point.
    By Lemma~\ref{lm:saddle-is-central}, \cDist is a \cd.

    The saddle point $(\cDist,R)$ satisfies the conditions of Lemma~\ref{lm:saddle_conditions}.
    Equations~\ref{eq:kkt_w_avg},~\ref{eq:kkt_r_sums_one} and~\ref{eq:kkt_r_geq_0} imply that $\cDist$ is a convex
    combination of the elements of~\setQ.
    If $R(i) > 0$, then Equation~\ref{eq:kkt_rmu_is_zero} implies that $\mu(i) = 0$.
    Substituting into Equation~\ref{eq:kkt_lambda} yields
    \[
        \kl[Q_i][\cDist] = \lambda.
    \]
    If $R(i) = 0$, then Equation~\ref{eq:kkt_mu_geq_zero} implies that $\mu(i) \geq 0$.
    Combining this with Equation~\ref{eq:kkt_lambda} gives
    \[
        \kl[Q_i][\cDist] \leq \lambda.
    \]

    For any probability distribution $C'$, we have
    \begin{align}\label{eq:cd-relation-to-lambda}
        \nonumber
        \kl[\cDist][C']
        &= \sum_i R(i)\left(\kl[Q_i][C'] - \kl[Q_i][\cDist] \right) \\
        \nonumber
        &= \sum_i \kl[Q_i][C']R(i) - \lambda \\
        &\leq \sFunc[C'] - \lambda.
    \end{align}
    Now suppose that $C'$ is also a central distribution of \setQ for $X$.
    Since \cDist and $C'$ are both central distributions,
    \[
        \sFunc[C'] = \sFunc[\cDist] = \lambda.
    \]
    Therefore, by non-negativity of the KL divergence, we have
    \[
        \kl[\cDist][C'] = 0,
    \]
    which implies \cDist and $C'$ are identical probability distributions.

    Thus, we have proved that \setQ has a unique central distribution satisfying the conditions of the theorem.
    We now prove the ``if'' direction.
    Assume that the conditions of the theorem hold for \cDist.
    Since \cDist is a convex combination of the elements of \setQ it is a probability distribution over $X$.
    By defining
    \[
        \mu(i) = \lambda - \kl[Q_i][\cDist],
    \]
    we have $\mu(i) \geq 0$ and $\mu(i)R(i) = 0$ for every $i$.
    We can verify that \cDist and $R$ satisfy the conditions of Lemma~\ref{lm:saddle_conditions} and therefore
    $\left( \cDist, R \right)$ is a saddle point for $\objf[]$.
    By Lemma~\ref{lm:saddle-is-central} \cDist is a \cd.
\end{proof}

\begin{corollary}
    \label{cor:cdiv-lambda-bound}
    Let \cDist be the \cd.
    For any probability distribution $P$, we have
    \begin{equation*}
        \myDiv[P][Q][(X)] \leq \sFunc[P(X)][\setQ] - \sFunc[\cDist(X)][\setQ].
    \end{equation*}
\end{corollary}

\begin{proof}
    By Theorem~\ref{th:centrality}, \cDist is unique,
    and therefore
    \[
        \myDiv[P][Q][] = \kl[\cDist].
    \]
    The result then follows immediately from Equation~\ref{eq:cd-relation-to-lambda}.
\end{proof}

\begin{theorem}
    \label{th:C-maximizes-g}
    The \cd\ is a maximizer of the function $g$ defined in Lemma~\ref{lm:saddle-is-maximizer}.
\end{theorem}

\begin{proof}
    By Lemmas ~\ref{lm:saddle-exists} and~\ref{lm:saddle-is-central}, there exists a saddle point $(\cDist,R)$
    of $\objf[]$, where \cDist is the unique \cd\ (Theorem~\ref{th:centrality}).
    Defining $C(X, I) = Q_I(X)R(I)$, Lemma~\ref{lm:saddle-is-maximizer} implies that \cDist is a maximizer
    of the function $g$.
\end{proof}

\begin{corollary}
    \label{cor:imperfection}
    Let \cDist be the \cd,
    and let $P$ be a convex combination of the elements of \setQ, i.e.,
    \[
        P(x) = \sum_i Q_i(x) R(i),
    \]
    where $R$ is a probability distribution.
    Then there exists an index $k$ in the support of $R$ such that
    \begin{equation*}
        \sFunc[\cDist(X)][\setQ] \geq \klx[Q_k].
    \end{equation*}
\end{corollary}

\begin{proof}
    We proceed by contradiction.
    Suppose that for every $k$ in the support of $R$, we have $\kl[Q_k] > \sFunc[\cDist]$.
    That implies
    \[
        g(P) > \sFunc[\cDist] = g(C),
    \]
    where $g$ is the function defined in
    Lemma~\ref{lm:saddle-is-maximizer}.
    Note that $g(P)$ is defined, because $P$ is a convex combination of the elements of \setQ.

    However, by Theorem~\ref{th:C-maximizes-g}, the distribution $C$ maximizes $g$, which contradicts the
    inequality $g(P) > g(C)$.
\end{proof}
