\begin{definition}[Probabilistic Model]
    We define a \emph{probabilistic model} \M as a pair $\left(\set{S}, \set{P}\right)$, where \set{S} is a sample
    space, and \set{P} is a set of probability measures on \set{S}.
\end{definition}

\begin{definition}[Divergence Measure]
    Let $\pairsX$ be a set of pairs of discrete random
    variables in a probabilistic model \model.
    We define the \emph{divergence measure of $P \in \set{P}$ for \set{X}} as follows:
    \begin{equation}
        \DxP = \max_{Q \in \set{P}} \Ex{Q}{\sum_i \Dcond},
        \label{eq:div_measure}
    \end{equation}
    where $\Ex{Q}{.}$ denotes expectation relative to $Q$ and each $\Dcond$ is the Kullback-Leibler divergence
    between $P(\rv{X}|\rval{E})$ and $Q(\rv{X}|\rval{E})$:
    \[
        \Dcond = \sum_{\rval{x}} Q(\xIe) \ln \frac{Q(\xIe)}{P(\xIe)}\cdot
    \]
    Moreover, we define \emph{the central distribution for \set{X}} to be a distribution $C_\set{X} \in \set{P}$ which
    minimizes the divergence measure for \set{X}:
    \begin{equation}
        C_\set{X} = \arg\pmin \DxP\cdot
        \label{eq:center}
    \end{equation}
\end{definition}

\begin{proposition}
    Using the linearity of expectation, we have:
    \begin{equation*}
        \DxP = \max_{Q \in \set{P}} \sumOverixe Q(\xe) \ln \frac{Q(\xIe)}{P(\xIe)}\cdot
        \label{eq:div_measure_full}
    \end{equation*}
\end{proposition}

\begin{definition}[Parametric Model]
    Let $\param$ be a random vector.
    A parametric model is a probabilistic model \pmodel, where for
    every $P \in \set{P}$ there is a prior probability $p$, such that for every $\rval[]{s} \in \set{s}$ we have:
    \begin{equation}
        \paramDist{\rval[]{s}},
        \label{eq:parametric}
    \end{equation}
    while $\pi$ is some known probability distribution.
\end{definition}

\begin{lemma}
    Let \pmodel be a parametric model, and $\pairsX$ be a set of discrete random variable pairs in \Mt.
    For every $P \in \set{P}$ we have:
    \begin{align}
        \DxP &\leq \tmax \DExparam\\  \label{eq:DxP_upper_bound}
        &= \tmax \Dparam\cdot
    \end{align}
\end{lemma}

\begin{proof}
    Let $R$ be an arbitrary distribution in \Mt, since the Kullback-Leibler divergence is always positive,
    its expected value is also positive, and for every parameter value $\paramValue$ we have:
    \[
        \Dparam[R] \geq 0\ \cdot
    \]
    Using this, we can derive the following inequality for every distribution $P$ and
    every parameter \vval{\theta}:
    \newcommand{\RP}{\sumOverixe \pi(\xeIt) \ln \frac{R(\xIe)}{P(\xIe)}}
    \[
        \RP \leq \Dparam\cdot
    \]
    Since this inequality holds for any $\paramValue$, the maximum of the right-hand side
    is also greater than or equal to the maximum of the left-hand side w.r.t $\paramValue$:
    \begin{equation}
        \tmax \RP \leq \tmax \Dparam\cdot
        \label{eq:ineq-theorem1}
    \end{equation}

    \Mt is a parametric model and $\paramDist[R]{\xe}$, while $r(\paramValue)$ is
    a probability distribution.
    Obviously for every $\paramValue$ we have:
    \[
        \RP \leq \tmax \RP\cdot
    \]
    Since $\prob [r]{\theta}$ is a distribution and is non-negative, we can multiply this inequality
    by $\prob [r]{\theta}$ and integrate.
    Notice that the right hand side is not a function of $\paramValue$ and it will remain unchanged
    because $\int \prob [r]{\theta} d\theta = 1$:
    \[
        \Ex{R}{\sum_i \Dcond[R]} \leq \tmax \RP\cdot
    \]
    Now by Inequality~\ref{eq:ineq-theorem1}, for \emph{every} distribution $P$ and $R$ we conlcude:
    \[
        \Ex{R}{\sum_i \Dcond[R]} \leq \tmax \Dparam,
    \]
    which proves the theorem.
\end{proof}

\begin{theorem}
    Let \pmodel be a parametric model and $\set{X}$ be a set of pairs of discrete random variables in \Mt.
    If for every value of $\paramValue$ there is a distribution $R \in \set{P}$ such that
    $R(\rval[]{s}) = \pi(\rval[]{s} \mid \paramValue)$ for every $\rval[]{s} \in \set{S}$,
    then for any $P \in \set{P}$ we have:
    \begin{equation}
        \DxP = \tmax \DExparam\cdot
        \label{eq:div_parametric}
    \end{equation}
    \label{th:div_for_parametric}
\end{theorem}

\begin{proof}
    For an arbitrary distribution $P \in \set{P}$, we let:
    \[
        \paramValue^* = \arg\tmax \DExparam\cdot
    \]
    In \Mt there should be a distribution $R \in \set{P}$ that equals $\pi(\,.\,|\paramValue^*)$, and we have:
    \[
        \Ex{R}{\sum_i \Dcond[R]} = \DExparam[\paramValue^*]\cdot
    \]
    On the other hand, we see:
    \[
        \max_{Q \in \set{P}} \Ex{Q}{\sum_i \Dcond} \geq \Ex{R}{\sum_i \Dcond[R]},
    \]
    Thus:
    \[
        \DxP \geq \tmax \DExparam\cdot
    \]
    This and Inequality~\ref{eq:DxP_upper_bound} proves the theorem.
\end{proof}

\begin{theorem}[Minimax theorem~\cite{Du1995}]
    Let $X \subset \mathbb {R}^{n}$ and $Y \subset \mathbb {R} ^{m}$ be compact convex sets.
    If $f:X\times Y\rightarrow \mathbb {R}$ is a continuous function that is concave-convex, i.e.

    \begin{itemize}
        \item $f(\cdot ,y):X\to \mathbb {R}$ is concave for fixed $y$, and
        \item $f(x,\cdot ):Y\to \mathbb {R}$ is convex for fixed $x$.
    \end{itemize}
    Then we have that:
    \begin{equation}
        \max_{x\in X}\min_{y\in Y}f(x, y) = \min_{y\in Y}\max_{x\in X}f(x,y)\ \cdot
        \label{eq:minimax}
    \end{equation}
\end{theorem}
