\begin{definition} [Sample Space]
    A sample space, denoted by \sampleSet, is the set of all possible outcomes of a random experiment.
    That is, the outcome of a random experiment can \emph{always} be represented by \emph{exactly one} element of \sampleSet.
\end{definition}

\begin{definition} [Action Space]
    Let \sampleSet be a sample space.
    The action space of \sampleSet, denoted by \actionSet, is the set of all functions from \sampleSet to \field{R}, where \field{R}
    is the field of real numbers.
\end{definition}

\begin{proposition}
    The action space equals $\field{R}^{\sampleSet}$, and is a vector space.
\end{proposition}

We can think of the members of \actionSet as hypothetical actions or gambles.
Each action associates an amount of loss with every possible outcome \inS.
This loss is quantified by a real number.
A negative loss indicates profit.

\begin{definition} [Sure Thing]
    The \emph{sure thing}, denoted by \sure[], is an element of \actionSet, such that $\sure[](s) = 1$ for every \inS.
\end{definition}

\begin{definition}
    Let \eventInS.
    The restriction to \event operator, denoted by \rstOp, is a linear operator on \actionSet,
    such that $\rstOp(\act)$ for any \inA{\act} is an element of \actionSet and
    \[
        \rstOp(\act)(s) =
        \begin{cases}
            \act(s)       & s \in \event, \\
            0       & s \in \sampleSet - \event \cdot
        \end{cases}
    \]
    We denote the range of \rstOp by \rstActionSet.
\end{definition}

\begin{proposition}
    \rstActionSet is an invariant subspace of \actionSet under \rstOp.
    If $\{\event[1], \event[2], \ldots \}$ is a partition of \sampleSet, then
    \[
        \actionSet = \rstActionSet[1] \oplus \rstActionSet[2] \oplus \ldots,
    \]
    where $\oplus$ indicates direct sum~\cite{axler}.
\end{proposition}

\begin{definition} [Linear Functional]
    Let \set{V} be a vector space over a field \field{F}.
    A linear functional on \set{V} is a linear map from \set{V} to \field{F}.
\end{definition}

\begin{definition} [Expectation]
    An \emph{expectation} for \sampleSet is a linear functional on \actionSet, denoted by \Expect, with the
    following properties:
    \begin{itemize}
        \item For the sure thing \sure, we have $\Expect[{\sure}] = 1$.
        \item For any \eventInS, we have $\Expect[\rstOp(\sure)] \geq 0$.
    \end{itemize}
\end{definition}

\begin{proposition}
    \Expect is an element of the dual space of \actionSet.
\end{proposition}

\begin{definition} [Conditional Expectation]
    Let \Expect be an expectation for \sampleSet and let \eventInS.
    \Expect conditioned on \event, denoted by \ExpectOnE, is a linear map on \sampleSet with the following properties:
    \begin{itemize}
        \item For the sure thing \sure, we have $\ExpectOnE[{\sure}] = 1$.
        \item For any \inA{\act}, and \eventInS,
        we have $\ExpectOnE[{\act}] = 0$ if and only if $\Expect[\rstOp(\act)] = 0$.
    \end{itemize}
\end{definition}

At this point, we have a complete framework for decision-making under uncertainty.
Interestingly, we do not need to explicitly define probability.
However, in order to present a more familiar framework, we provide a definition of conventional probability.

\begin{definition} [Probability]
    We define the probability of \eventInS as follows:
    \[
        P(\event) = \Expect[\rstOp(\sure)] \cdot
    \]
\end{definition}

We now will establish the familiar relationship between conditional expectation and normal expectation.

\begin{lemma}
    \label{lm:scale}
    Let \F and \G be two linear functionals defined on a vector space \set{V}.
    \F and \G have the same null space if and only if $\F = \lambda \G$ for some $\lambda \neq 0$.
\end{lemma}

\begin{proof}
    If $\F = \lambda \G$, $\F[v] = 0$ if and only if $\G[v] = 0$ for every $v \in \set{V}$.
    This proves the if part.

    We prove the only if part by contradiction.
    Assume $\F \neq \lambda \G$.
    Then there exist $u,v \in \set{v}$ such that
    \begin{equation*}
        \F[u] \G[v] - \F[v] \G[u] \neq 0 \cdot
    \end{equation*}
    Let $w = \F[u]v - \F[v]u$.
    Since \set{V} is a vector space $w \in \set{V}$.
    \F and \G are linear.
    Hence, $\F[w] = 0$ and $\G[w] = \F[u] \G[v] - \F[v] \G[u] \neq 0$ and the null spaces of \F and \G are not
    equal.
    This contradiction completes the proof.
\end{proof}

\begin{theorem}
    Let \inA{\act} and \eventInS, where \sampleSet is a sample space.
    If \rstOp is the restriction operator to \event, then we have
    \begin{equation*}
        \ExpectOnE[\act] = \frac{\Expect[\rstOp(\act)]}{P(\event)} \cdot
    \end{equation*}
\end{theorem}

\begin{proof}
    \renewcommand{\a}{\act[]}
    \renewcommand{\aa}{\act[]'}
    For an arbitrary \inA{\a}, let $\aa = \a - \rstOp(\a)$.
    Clearly, $\rstOp(\aa)$ is the zero function, and therefore $\Expect[\rstOp(\aa)] = 0$.
    Since \actionSet is a vector space we have \inA{\aa}, and the definition of conditional expectation
    implies $\ExpectOnE[\aa] = 0$.
    Therefore
    \begin{equation}
        \ExpectOnE[\a] - \ExpectOnE[\rstOp(\a)] = 0 \cdot
        \label{eq:exp_cond_theorem}
    \end{equation}

    Let \rstActionSet represent the range of \rstOp.
    \rstActionSet is a subspace of \actionSet.
    By the definition of conditional expectation, functionals obtained by the restrictions of \Expect and \ExpectOnE
    to \rstActionSet have the same null space.
    That implies by Lemma~\ref{lm:scale} for every \inA{\a} we have
    \begin{equation*}
        \ExpectOnE[\rstOp(\a)] = \lambda \Expect[\rstOp(\a)] \cdot
    \end{equation*}
    Thus, by Equation~\ref{eq:exp_cond_theorem}
    \begin{equation*}
        \ExpectOnE[\a] = \lambda \Expect[\rstOp(\a)] \cdot
    \end{equation*}

    On the other hand, for the sure thing \sure, we have
    \begin{equation*}
        \lambda = \frac{\ExpectOnE[\sure]}{\Expect[\rstOp(\sure)]} = \frac{1}{P(\event)} \cdot
    \end{equation*}
    This proves the theorem.
\end{proof}

\begin{definition} [Dual Map]
    Let \set{a} and \set{b} be two sets.
    Suppose $\maps{a}{b}$ is a map from \set{a} to \set{b}.
    The dual map of $M$ is the map $\mapsDual{b}{a}$, defined for each $f \in \field{R}^{\set{b}}$ by
    \begin{equation*}
        \dualMap(f) = f \circ M \cdot
    \end{equation*}
\end{definition}

\begin{proposition}
    The dual map is a linear map.
\end{proposition}

\begin{theorem}
    Let \sampleSet be a sample space.
    Suppose  $\maps{\sampleSet}{\sampleSetAlt}$ is a map from \sampleSet to another set \sampleSetAlt.
    If \Expect is an expectation for \sampleSet,
    then $\ExpAlt = \dualdualMap(\Expect)$ is an expectation for \sampleSetAlt,
    where \dualdualMap is the dual map of \dualMap, and \dualMap is the dual map of $M$.
\end{theorem}

\begin{proof}
    \dualdualMap is a dual map and therefore is linear.

    By definition for every $\act[\sampleSetAlt] \in \actionSet[\sampleSetAlt]$ we have
    \begin{equation*}
        \ExpAlt[{\act[\sampleSetAlt]}] = \Expect \circ \act[\sampleSetAlt] \circ M \cdot
    \end{equation*}
    We can easily verify that
    \begin{equation*}
        \sureAlt \circ M = \sure \cdot
    \end{equation*}
    Therefore, $\ExpAlt[\sureAlt] = 1$.

    On the other hand for any $\eventAlt \subseteq \sampleSetAlt$, we have
    \[
        \rstOp[\eventAlt](\sureAlt) \circ M = \rstOp(\sure),
    \]
    where
    \[
        \event = \{ s \in \sampleSet : M(s) \in \eventAlt \} \cdot
    \]
    This implies  $\ExpAlt[{\rstOp[\eventAlt](\sureAlt)}] \geq 0$.
\end{proof}
