The results of the previous subsection were established under the assumption that \setQ is finite.
Many of the underlying constructions, however, naturally extend to parametric families
in which \setQ is indexed by a finite-dimensional parameter vector \param:
\[
    \setQ = \left\{ Q_{\theta} \mid \theta \in \operatorname{val}(\param) \right\}.
\]
In this setting, \param may be regarded as a random variable, with each $Q_{\paramVal}$ interpreted as a conditional
distribution~$Q(\xIt)$.
The weight function $R(i)$ appearing in the finite case is then replaced by a prior distribution over~\param.

A rigorous development of this extension requires additional measure-theoretic and regularity assumptions and is
beyond the scope of the present paper.

\begin{definition}[Central Prior] \label{def:central-prior}
    Let $X$ be a random variable, and let \setQ be a set of probability distributions over $X$,
    indexed by a parameter \param.
    A nonnegative measure $c$ on \param is called a central prior of \setQ for $X$ if,
    \[
        \cDist(x) = \mrg
    \]
    is a central distribution of \setQ for $X$.
\end{definition}

\begin{definition}[Interior Distribution] \label{def:interior}
    Let \setQ be a set of probability distributions over a random variable $X$.
    A distribution $R$ is called an \emph{interior distribution} of \setQ if and only if
    the sets \setQ and $\setQ \cup \{R\}$ have the same central distributions for $X$.
\end{definition}

\begin{theorem}
    Suppose that the conditions of Theorem~\ref{th:centrality} hold,
    and let \cDist denote the unique central distribution of \setQ (as defined therein).
    If for some $Q_k \in \setQ$
    \[
        \klx[Q_k][\cDist] < \lambda,
    \]
    then $Q_k$ is an interior distribution of \setQ.
\end{theorem}

\begin{proof}
    Since $\kl[Q_k][\cDist] < \lambda$, we must have $R(k) = 0$.
    Hence, \cDist is a convex combination of elements of $\setQ \setminus{\{Q_k\}}$ and
    the conditions of Theorem~\ref{th:centrality} remain satisfied for $\setQ \setminus{\{Q_k\}}$.
    Thus, \cDist is the unique central distribution of $\setQ \setminus{\{Q_k\}}$ and \setQ.
\end{proof}

\begin{theorem}
    Let $X$ be a random variable and let \setQ be a set of probability distributions over $X$.
    Any distribution that can be expressed as a convex combination of elements of \setQ is an
    interior distribution of \setQ.
\end{theorem}

\begin{proof}
    Let $R = \sum_i \alpha_i Q_i$, where $Q_i \in \setQ$, $\alpha_i \geq 0$, and $\sum_i \alpha_i = 1$.

    Fix an arbitrary probability distribution $P$.
    By the convexity of KL divergence in its first argument, we have:
    \[
        \kl[R] \leq \sum_i \alpha_i \kl[Q_i].
    \]
    Note that the right-hand side is a weighted average of the KL divergences.
    So, there must be at least one index $k$ such that
    \[
        \kl[R] \leq \kl[Q_k].
    \]
    Since $Q_k \in \setQ$, it follows that
    \[
        \sFunc[P][\setQ] = \sFunc[P][\setQ \cup \{R\}]
    \]
    for every probability distribution $P$.
    Therefore, \setQ and $\setQ \cup \{R\}$ have the same central distributions.
\end{proof}

One important consequence of this theorem is that, in order to identify a central distribution within a parametric
model, it suffices to restrict attention to the set of likelihood distributions induced by different values of the
model parameters.

\begin{theorem}
    \label{th:estimator-to-var}
    Let $X$ and $T$ be two random variables, and let \setQ be a set of distributions over the pair $(X,T)$.
    Assume that for every \inSet{Q}, we have
    \[
        Q(X, T) = R(\XIT) Q(T),
    \]
    where $R$ is a fixed conditional distribution.
    If $C$ is a central distribution of \setQ for $T$, then $R(\XIT)C(T)$ is a central
    distribution of \setQ for $(X,T)$.
\end{theorem}

\begin{proof}
    Let $C$ be a central distribution of \setQ for $T$.
    For any arbitrary probability distribution $P$, the definition of a central distribution implies
    \begin{equation}
        \label{eq:th5-proof}
        \sFunc[R(\XIT) C(T)] \leq \sFunc[R(\XIT) P(T)].
    \end{equation}

    Since the KL divergence is nonnegative, for every probability distribution $P$ and every \inSet{Q} we have
    \[
        \sum_t \Dkl{R(X \mid t)}{P(X \mid t)} Q(t) \geq 0.
    \]
    That implies
    \[
        \Dkl{Q(X,T)}{R(\XIT) P(T)} \leq \Dkl{Q(X,T)}{P(X,T)}.
    \]
    Taking the supremum over \inSet{Q} preserves the inequality, since it holds pointwise for every \inSet{Q}:
    \[
        \sFunc[R(\XIT) P(T)] \leq \sFunc[P(X,T)].
    \]
    Combining this inequality with Equation~\ref{eq:th5-proof}, we conclude that $R(\XIT)C(T)$ is a
    central distribution of \setQ for $(X,T)$.
\end{proof}

The following theorem provides a sufficient fixed-point characterization of central priors
and highlights a formal connection between the central distribution framework and Bernardo's reference
prior methodology~\cite{BERNARDO200517}.

\begin{theorem}
    Let $X$ be a discrete random variable, and let \setQ be a set of probability distributions over $X$, indexed by a
    finite-dimensional vector \param.
    Let \cPrior be a (possibly improper) prior distribution for \param such that for every \paramVal we have
    \begin{equation}
        \label{eq:central-prior}
        \cPrior(\paramVal) \propto \exp \left\{  \sum_{x} \like(\xIt) \ln \cPrior(\tIx) \right\}.
    \end{equation}
    Then, \cPrior is a central prior of \setQ for $X$.
\end{theorem}

\begin{proof}
    \newcommand{\cp}{\cPrior(\paramVal{})}
    For every \paramVal, since $Q_{\paramVal} \in \setQ$ is a probability distribution,
    there exists $x$ such that $Q_{\paramVal}(x) > 0$.
    Equation~\ref{eq:central-prior} therefore implies $\cPrior(\tIx) > 0$,
    and consequently $\cp > 0$.

    Using Equation~\ref{eq:central-prior} and $\cp > 0$, we obtain
    \begin{align*}
        \cp
        &= \exp \left\{-\lambda +  \sum_{x} \like(\xIt) \ln \frac{\like(\xIt) \cp}{\cDist(x)} \right\} \\
        &= \exp \left\{-\lambda +  \klx[Q_{\paramVal{}}][C] + \ln \cp \right\} \\
        &= \cp \exp \left\{-\lambda +  \klx[Q_{\paramVal{}}][C] \right\},
    \end{align*}
    and it follows that
    \[
        \klx[Q_{\paramVal{}}][C] = \lambda.
    \]
    Since this holds for every \paramVal, and $\lambda$ is constant,
    a parametric extension of Theorem~\ref{th:centrality} implies that $\cDist$,
    the marginal distribution induced by \cPrior, is a central distribution of \setQ for~$X$.
    Therefore, by Definition~\ref{def:central-prior}, \cPrior is a central prior of \setQ for~$X$.
\end{proof}

Although this theorem is proved directly from Theorem~\ref{th:centrality}, Lemma~\ref{lm:saddle-is-maximizer}
suggests an alternative route to the connection with reference priors.
Since reference priors are defined by maximizing missing information,
the optimization characterization provided by Lemma~\ref{lm:saddle-is-maximizer} also offers another perspective on
this relationship.
